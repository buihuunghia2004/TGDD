import { OffsetPaginatedDto } from '@/common/dto/offset-pagination/paginated.dto';
import { Uuid } from '@/common/types/common.type';
import { ApiAuth } from '@/decorators/http.decorators';
import {
  Body,
  Controller,
  Delete,
  Get,
  HttpStatus,
  Param,
  ParseUUIDPipe,
  Patch,
  Post,
  Query,
  UseGuards,
} from '@nestjs/common';
import { ApiParam, ApiTags } from '@nestjs/swagger';
import { AuthGuard } from '@/guards/auth.guard';
import { yxxService } from './xxx.service';
import { yxxResDto } from './dto/xxx.res.dto';
import { CreateyxxReqDto } from './dto/create-xxx.req.dto';
import { ListyxxReqDto } from './dto/list-xxx.req.dto';
import { UpdateyxxReqDto } from './dto/update-xxx.req.dto';

@ApiTags('xxxs')
@UseGuards(AuthGuard)
@Controller({
  path: 'xxxs',
  version: '1',
})
export class yxxController {
  constructor(private readonly xxxService: xxxService) {}

  @Post()
  @ApiAuth({
    type: yxxResDto,
    summary: 'Create xxx',
    statusCode: HttpStatus.CREATED,
  })
  async createyxx(
    @Body() createyxxDto: CreateyxxReqDto,
  ): Promise<yxxResDto> {
    return await this.xxxService.create(createyxxDto);
  }

  @Get()
  @ApiAuth({
    type: yxxResDto,
    summary: 'List xxxs',
    isPaginated: true,
  })
  async findAllyxxs(
    @Query() reqDto: ListyxxReqDto,
  ): Promise<OffsetPaginatedDto<yxxResDto>> {
    return await this.xxxService.findAll(reqDto);
  }

  @Get(':id')
  @ApiAuth({ type: yxxResDto, summary: 'Find xxx by id' })
  @ApiParam({ name: 'id', type: 'String' })
  async findyxx(@Param('id', ParseUUIDPipe) id: Uuid): Promise<yxxResDto> {
    return await this.xxxService.findOne(id);
  }

  @Patch(':id')
  @ApiAuth({ type: yxxResDto, summary: 'Update xxx' })
  @ApiParam({ name: 'id', type: 'String' })
  updateyxx(
    @Param('id', ParseUUIDPipe) id: Uuid,
    @Body() reqDto: UpdateyxxReqDto,
  ) {
    return this.xxxService.update(id, reqDto);
  }

  @Delete(':id')
  @ApiAuth({
    summary: 'Delete xxx',
    errorResponses: [400, 401, 403, 404, 500],
  })
  @ApiParam({ name: 'id', type: 'String' })
  removeyxx(@Param('id', ParseUUIDPipe) id: Uuid) {
    return this.xxxService.remove(id);
  }
}
